  <script>
    document.addEventListener('DOMContentLoaded', () => {
      // === 1. БАЗА ДАННЫХ ===
      const recipesDB = [

          steps: ["Залейте хлопья молоком.", "Варите на медленном огне 5-7 мин, помешивая.", "Нарежьте яблоко кубиками, добавьте в кашу."] },

        { id: 4, name: "Греческий салат", category: "ужин", difficulty: "простой",
          ingredients: [{ name: "Помидоры", weight_g: 200, cal: 20, prot: 0.6, fat: 0.2, carb: 4.2 }, { name: "Огурцы", weight_g: 150, cal: 15, prot: 0.8, fat: 0.1, carb: 2.8 }, { name: "Сыр фета", weight_g: 100, cal: 264, prot: 14, fat: 21, carb: 4.1 }, { name: "Оливки", weight_g: 50, cal: 145, prot: 1, fat: 15, carb: 3 }],

          ingredients: [{ name: "Говядина", weight_g: 300, cal: 250, prot: 26, fat: 16, carb: 0 }, { name: "Свекла", weight_g: 150, cal: 42, prot: 1.5, fat: 0.1, carb: 9.9 }, { name: "Капуста", weight_g: 200, cal: 27, prot: 1.8, fat: 0.1, carb: 4.7 }, { name: "Картофель", weight_g: 250, cal: 77, prot: 2, fat: 0.4, carb: 16.3 }],
          steps: ["Сварите бульон из говядины (1.5 часа).", "Натрите свеклу, потушите с уксусом и томатом.", "Добавьте в бульон картофель и капусту.", "Через 10 мин добавьте зажарку, варите 5 мин."] }
      ];

      // === 2. ВСПОМОГАТЕЛЬНЫЕ ФУНКЦИИ ===
      const normalize = str => str.toLowerCase().replace(/[^а-яёa-z0-9\s]/g, '').trim();
      const fuzzyMatch = (query, target) => {
        const q = normalize(query), t = normalize(target);
        if (q === "" || t === "") return false;
        if (t.includes(q) || q.includes(t)) return true;
        if (q.length <= 15) {
          let dist = 0, len1 = q.length, len2 = t.length;
          const dp = Array(len1 + 1).fill().map(() => Array(len2 + 1).fill(0));
          for (let i = 0; i <= len1; i++) dp[i][0] = i;
          for (let j = 0; j <= len2; j++) dp[0][j] = j;
          for (let i = 1; i <= len1; i++) for (let j = 1; j <= len2; j++) {
            const cost = q[i-1] === t[j-1] ? 0 : 1;
            dp[i][j] = Math.min(dp[i-1][j] + 1, dp[i][j-1] + 1, dp[i-1][j-1] + cost);
          }
          return dp[len1][len2] <= 2;
        }
        return false;
      };
      const calcKBZU = (recipe) => {
        let totalCal=0, totalProt=0, totalFat=0, totalCarb=0, totalWeight=0;
        recipe.ingredients.forEach(ing => {
          const w = ing.weight_g;
          totalCal += (ing.cal * w)/100; totalProt += (ing.prot * w)/100;
          totalFat += (ing.fat * w)/100; totalCarb += (ing.carb * w)/100; totalWeight += w;
        });
        return { cal: Math.round((totalCal/totalWeight)*100), prot: Math.round((totalProt/totalWeight)*100), fat: Math.round((totalFat/totalWeight)*100), carb: Math.round((totalCarb/totalWeight)*100) };
      };

      // === 3. ЛОГИКА ПОИСКА ===
      const findRecipes = () => {
        const rawInput = document.getElementById('ingredientsInput').value;
        const userIngredients = rawInput.split(',').map(s => s.trim()).filter(s => s);
        if (userIngredients.length === 0) { alert("Введите хотя бы один продукт!"); return; }
        const diffFilter = document.getElementById('difficultySelect').value;
        const catFilter = document.getElementById('categorySelect').value;
        const calLimit = parseInt(document.getElementById('calorieLimit').value) || null;
        const results = recipesDB.filter(recipe => {
          if (diffFilter !== 'all' && recipe.difficulty !== diffFilter) return false;
          if (catFilter !== 'all' && recipe.category !== catFilter) return false;
          const kbjU = calcKBZU(recipe);
          if (calLimit && kbjU.cal > calLimit) return false;
          let missing = [];
          recipe.ingredients.forEach(ing => { if (!userIngredients.some(u => fuzzyMatch(u, ing.name))) missing.push(ing.name); });
          return missing.length <= 2;
        });
        renderResults(results, userIngredients);
        saveHistory(rawInput, diffFilter, catFilter, calLimit);
      };

      // === 4. ОТРИСОВКА ===
      const renderResults = (recipes, userIngredients) => {
        const container = document.getElementById('results');
        container.innerHTML = '';
        if (recipes.length === 0) { container.innerHTML = '<div class="empty-state">😔 Рецепты не найдены. Добавьте больше продуктов или ослабьте фильтры.</div>'; return; }
        recipes.forEach(r => {
          const kbjU = calcKBZU(r);
          let missing = [];
          r.ingredients.forEach(ing => { if (!userIngredients.some(u => fuzzyMatch(u, ing.name))) missing.push(ing.name); });
          const card = document.createElement('div');
          card.className = 'recipe-card';
          card.innerHTML = `
            <div class="recipe-title"><span>${r.name}</span><span class="badge">${r.difficulty}</span></div>
            <div style="font-size:14px; color:var(--text-light); margin-bottom:6px;">🍽 ${r.category}</div>
            <div class="kbzu-box"><b>КБЖУ на 100г:</b> ${kbjU.cal} ккал | Б: ${kbjU.prot}г | Ж: ${kbjU.fat}г | У: ${kbjU.carb}г</div>
            <ul class="ingredients-list">${r.ingredients.map(i => `<li><b>${i.name}</b> — ${i.weight_g}г (${i.cal} ккал/100г)</li>`).join('')}</ul>
            ${missing.length > 0 ? `<div class="missing">⚠️ Не хватает: ${missing.join(', ')}</div>` : '<div style="color:var(--success); font-weight:500; margin-top:6px;">✅ Все продукты есть!</div>'}
            <div style="margin-top:10px; font-weight:500;">📝 Приготовление:</div>
            <ol class="steps">${r.steps.map(s => `<li>${s}</li>`).join('')}</ol>`;
          container.appendChild(card);
        });
      };

      // === 5. ИСТОРИЯ ===
      const saveHistory = (ing, diff, cat, cal) => {
        let history = JSON.parse(localStorage.getItem('recipeHistory') || '[]');
        history = history.filter(h => h.ing !== ing).slice(0, 9);
        history.unshift({ ing, diff, cat, cal, time: new Date().toLocaleTimeString() });
        localStorage.setItem('recipeHistory', JSON.stringify(history));
        renderHistory();
      };
      const renderHistory = () => {
        const list = document.getElementById('historyList');
        const history = JSON.parse(localStorage.getItem('recipeHistory') || '[]');
        list.innerHTML = '';
        if (history.length === 0) { list.innerHTML = '<span style="color:var(--text-light); font-size:14px;">История пуста</span>'; return; }
        history.forEach(h => {
          const tag = document.createElement('span');
          tag.className = 'history-tag';
          tag.textContent = `${h.ing.substring(0, 20)}${h.ing.length>20?'...':''} (${h.cat})`;
          tag.onclick = () => {
            document.getElementById('ingredientsInput').value = h.ing;
            document.getElementById('difficultySelect').value = h.diff;
            document.getElementById('categorySelect').value = h.cat;
            if (h.cal) document.getElementById('calorieLimit').value = h.cal;
            findRecipes();
          };
          list.appendChild(tag);
        });
      };

      // === 6. ИНИЦИАЛИЗАЦИЯ ===
      document.getElementById('searchBtn').addEventListener('click', findRecipes);
      renderHistory();
    });

      });
    }
